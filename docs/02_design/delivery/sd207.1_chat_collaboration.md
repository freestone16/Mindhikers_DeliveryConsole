## 三、技术架构

### 3.1 数据流

```
用户输入 → ChatPanel → Socket
                      ↓
              ┌───────┴───────┐
              │ 意图识别       │
              │ (关键词/LMM)   │
              └───────┬───────┘
                      ↓
        ┌─────────────┴─────────────┐
        ↓                           ↓
   【普通对话】                【修改请求】
        ↓                           ↓
  流式返回 ChatPanel      1. 返回简短确认
                             2. 执行修改
                             3. 推送到专家模块
                                     ↓
                             专家表格更新 ✅
```

### 3.2 Socket 事件

| 事件名 | 方向 | 说明 |
|--------|------|------|
| `chat-stream` | Client → Server | 发送消息 |
| `chat-chunk` | Server → Client | 流式返回 |
| `chat-done` | Server → Client | 对话完成 |
| `expert-data-update` | Server → Client | **新增** 专家数据更新 |
| `chat-confirmation` | Server → Client | **新增** 修改确认响应 |

## 四、核心代码设计

### 4.1 后端：意图识别

```typescript
// server/chat.ts

interface ChatIntent {
    type: 'chat' | 'modify';
    confidence: number;
    target?: {
        expertId: string;
        action: ModifyAction;
        targetId?: string;
        payload?: any;
    };
}

type ModifyAction = 
    | 'update_script_text'
    | 'update_cta'
    | 'update_hook'
    | 'batch_update'
    | 'regenerate';

// 关键词模式
const MODIFY_PATTERNS = [
    { pattern: /第(\d+).*脚本.*[改修短换]/, action: 'update_script_text' },
    { pattern: /CTA.*[改修]/, action: 'update_cta' },
    { pattern: /钩子.*[换改]/, action: 'update_hook' },
    { pattern: /全部.*[改修]/, action: 'batch_update' },
    { pattern: /重新生成/, action: 'regenerate' },
];

export function parseIntent(
    userMessage: string, 
    expertId: string,
    context?: { scripts?: any[] }
): ChatIntent {
    // 1. 关键词匹配
    for (const { pattern, action } of MODIFY_PATTERNS) {
        const match = userMessage.match(pattern);
        if (match) {
            return {
                type: 'modify',
                confidence: 0.8,
                target: {
                    expertId,
                    action,
                    targetId: match[1] ? `script-${match[1]}` : undefined,
                    payload: { userMessage }
                }
            };
        }
    }
    
    // 2. 默认为普通对话
    return { type: 'chat', confidence: 1.0 };
}
```

### 4.2 后端：修改执行器

```typescript
// server/chat-modifications.ts

export async function executeModification(
    projectRoot: string,
    target: ChatIntent['target'],
    context: { scripts: ShortScript[] }
): Promise<{ success: boolean; data?: any; error?: string }> {
    
    if (!target) return { success: false, error: 'No target' };
    
    switch (target.action) {
        case 'update_script_text': {
            const scriptId = target.targetId;
            // 调用 LLM 生成修改后的脚本
            const newScriptText = await generateModifiedScript(
                context.scripts.find(s => s.id === scriptId),
                target.payload.userMessage
            );
            return {
                success: true,
                data: { scriptId, updates: { scriptText: newScriptText } }
            };
        }
        
        case 'update_cta': {
            // ...
        }
        
        default:
            return { success: false, error: 'Unknown action' };
    }
}
```

### 4.3 后端：Socket 事件扩展

```typescript
// server/index.ts

socket.on('chat-stream', async ({ messages, expertId, projectId }) => {
    const lastMessage = messages[messages.length - 1].content;
    const intent = parseIntent(lastMessage, expertId);
    
    if (intent.type === 'modify' && intent.confidence > 0.7) {
        // === 修改请求模式 ===
        
        // 1. 返回简短确认
        socket.emit('chat-confirmation', { 
            expertId, 
            message: '✅ 已收到修改请求，正在处理...' 
        });
        socket.emit('chat-done', { expertId });
        
        // 2. 执行修改
        const projectRoot = getProjectRoot(projectId);
        const currentData = loadExpertData(projectRoot, expertId);
        const result = await executeModification(projectRoot, intent.target, currentData);
        
        if (result.success) {
            // 3. 推送更新到专家模块
            socket.emit('expert-data-update', {
                expertId,
                action: intent.target!.action,
                data: result.data
            });
            
            // 4. 广播给所有客户端（包括发送者）
            io.emit('expert-data-update', {
                expertId,
                action: intent.target!.action,
                data: result.data
            });
        } else {
            socket.emit('chat-error', { 
                expertId, 
                error: `修改失败: ${result.error}` 
            });
        }
        
    } else {
        // === 普通对话模式（现有流程） ===
        const config = loadConfig();
        // ... 现有代码
    }
});
```
