## Feature 0: 进程健壮性升级 (Infrastructure)

> **优先级**: P0 - 基础设施，必须先实施
> **理由**: 当前开发环境启动/重启体验差，残留进程导致端口占用，严重影响开发效率

### 现状分析

| 问题 | 根因 | 影响 |
|------|------|------|
| 残留 tsx watch 进程 | `concurrently` 没有正确清理子进程 | 端口持续占用，无法重启 |
| 端口被占用 (3002, 5173) | 没有启动前检查 | 启动失败或运行错误版本 |
| 子进程泄漏 | `tsx watch`、`chokidar`、`spawn` 的进程树没有正确清理 | 资源浪费、状态不一致 |
| 无进程管理 | 开发环境纯 `npm run dev`，生产环境无管理 | 崩溃后无法自动恢复 |
| 无 graceful shutdown | Ctrl+C 时直接杀死进程，不清理资源 | 可能导致数据损坏 |

#### [NEW] scripts/predev.js - 启动前检查

```javascript
#!/usr/bin/env node
/**
 * 开发启动前检查脚本
 * 1. 检查端口占用
 * 2. 检查 .env 文件
 * 3. 检查必需目录
 */

import http from 'http';
import fs from 'fs';
import path from 'path';
import { fileURLToPath } from 'url';

const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);
const ROOT = path.join(__dirname, '..');

const PORTS = [3002, 5173];

async function checkPort(port) {
  return new Promise((resolve) => {
    const server = http.createServer();
    server.listen(port, () => {
      server.once('close', () => resolve(true));
      server.close();
    });
    server.on('error', () => resolve(false));
  });
}

function checkEnv() {
  const envPath = path.join(ROOT, '.env');
  if (!fs.existsSync(envPath)) {
    console.warn('⚠️  .env 文件不存在，使用默认配置');
    return false;
  }
  return true;
}

async function main() {
  console.log('🔍 启动前检查...\n');

  // 检查 .env
  const envOk = checkEnv();
  if (envOk) console.log('✅ .env 配置存在');

  // 检查端口
  let portsFree = true;
  for (const port of PORTS) {
    const free = await checkPort(port);
    if (free) {
      console.log(`✅ 端口 ${port} 可用`);
    } else {
      console.error(`❌ 端口 ${port} 被占用！请运行 npm run dev:clean 清理`);
      portsFree = false;
    }
  }

  if (!portsFree) {
    console.log('\n💡 提示：运行 npm run dev:force 强制清理并启动');
    process.exit(1);
  }

  console.log('\n✨ 所有检查通过，准备启动开发服务器...');
}

main().catch(console.error);
```

#### [NEW] scripts/kill-dev.js - 智能清理脚本

```javascript
#!/usr/bin/env node
/**
 * 开发环境进程清理脚本
 * 清理所有相关进程：
 * - tsx watch
 * - vite
 * - node (匹配特定模式)
 * - PM2 进程（开发时不应该有，但以防万一）
 */

import { spawn } from 'child_process';

// 要清理的进程模式
const PATTERNS = [
  'tsx watch.*server/index',
  'vite.*--host',
  'node.*server/index',
];

const PORTS = [3002, 5173];

function runCommand(cmd, args) {
  return new Promise((resolve, reject) => {
    const proc = spawn(cmd, args, { stdio: 'pipe' });
    let stdout = '';
    let stderr = '';

    proc.stdout.on('data', (data) => stdout += data);
    proc.stderr.on('data', (data) => stderr += data);

    proc.on('close', (code) => {
      if (code === 0) resolve(stdout);
      else reject(new Error(stderr));
    });
  });
}

async function killByPattern(pattern) {
  try {
    const output = await runCommand('pgrep', ['-fl', pattern]);
    if (!output.trim()) return [];

    const lines = output.trim().split('\n');
    const pids = lines.map(line => line.split(/\s+/)[0]).filter(Boolean);

    for (const pid of pids) {
      try {
        process.kill(pid, 'SIGTERM');
        console.log(`  ✓ 已终止进程 ${pid} (${pattern})`);
      } catch (e) {
        // 进程可能已经不存在
      }
    }

    // 等待后用 SIGKILL 清理顽固进程
    await new Promise(r => setTimeout(r, 500));
    for (const pid of pids) {
      try {
        process.kill(pid, 0); // 检查进程是否还存在
        process.kill(pid, 'SIGKILL');
        console.log(`  ✓ 强制终止进程 ${pid}`);
      } catch (e) {}
    }

    return pids;
  } catch (e) {
    return [];
  }
}

async function killByPort(port) {
  try {
    const output = await runCommand('lsof', ['-ti', `:${port}`]);
    if (!output.trim()) return [];

    const pids = output.trim().split('\n').filter(Boolean);
    for (const pid of pids) {
      try {
        process.kill(pid, 'SIGTERM');
        console.log(`  ✓ 已终止占用端口 ${port} 的进程 ${pid}`);
      } catch (e) {}
    }
    return pids;
  } catch (e) {
    return [];
  }
}

async function main() {
  console.log('🧹 清理开发环境进程...\n');

  let totalKilled = 0;

  // 按模式清理
  console.log('清理匹配的进程:');
  for (const pattern of PATTERNS) {
    const killed = await killByPattern(pattern);
    totalKilled += killed.length;
  }

  // 按端口清理
  console.log('\n清理占用端口的进程:');
  for (const port of PORTS) {
    const killed = await killByPort(port);
    totalKilled += killed.length;
  }

  // 清理 PM2（开发环境不应该有）
  try {
    await runCommand('pm2', ['kill']);
    console.log('  ✓ 已停止所有 PM2 进程');
    totalKilled++;
  } catch (e) {}

  console.log(`\n✨ 共清理 ${totalKilled} 个进程`);
}

main().catch(console.error);
```

#### [NEW] scripts/check-port.js - 端口检查工具

```javascript
#!/usr/bin/env node
import http from 'http';

async function checkPort(port) {
  return new Promise((resolve) => {
    const server = http.createServer();
    server.listen(port, () => {
      server.once('close', () => resolve({ port, status: 'free', process: null }));
      server.close();
    });
    server.on('error', async () => {
      // 找到占用端口的进程
      const { spawn } = await import('child_process');
      try {
        const proc = spawn('lsof', ['-ti', `:${port}`]);
        let output = '';
        proc.stdout.on('data', (d) => output += d);
        await new Promise(r => proc.on('close', r));
        resolve({ port, status: 'occupied', process: output.trim() });
      } catch (e) {
        resolve({ port, status: 'occupied', process: null });
      }
    });
  });
}

async function main() {
  const ports = [3002, 5173];
  const results = [];

  for (const port of ports) {
    const result = await checkPort(port);
    results.push(result);
    const icon = result.status === 'free' ? '✅' : '❌';
    const processInfo = result.process ? ` (PID: ${result.process})` : '';
    console.log(`${icon} 端口 ${port}: ${result.status}${processInfo}`);
  }

  const allFree = results.every(r => r.status === 'free');
  process.exit(allFree ? 0 : 1);
}

main().catch(console.error);
```

```javascript
#!/usr/bin/env node
import { spawn } from 'child_process';
import http from 'http';

async function checkPort(port) {
  return new Promise((resolve) => {
    const server = http.createServer();
    server.listen(port, () => {
      server.once('close', () => resolve(true));
      server.close();
    });
    server.on('error', () => resolve(false));
  });
}

async function main() {
  const FRONTEND_PORT = 5173;
  const BACKEND_PORT = 3002;

  // 检查端口
  const frontendFree = await checkPort(FRONTEND_PORT);
  const backendFree = await checkPort(BACKEND_PORT);

  if (!frontendFree || !backendFree) {
    console.error('❌ 端口被占用，请先运行 npm run dev:clean');
    process.exit(1);
  }

  console.log('🚀 启动预览服务器...\n');

  // 启动后端
  const backend = spawn('node', ['dist/server/index.js'], {
    stdio: 'inherit',
    env: { ...process.env, NODE_ENV: 'production' }
  });
  console.log(`✅ 后端服务器启动: http://localhost:${BACKEND_PORT}`);

  // 启动前端
  const frontend = spawn('npx', ['vite', 'preview', '--host'], {
    stdio: 'inherit',
    env: { ...process.env, NODE_ENV: 'production' }
  });

  // 优雅关闭
  const shutdown = () => {
    console.log('\n📴 正在关闭服务器...');
    backend.kill('SIGTERM');
    frontend.kill('SIGTERM');
    setTimeout(() => {
      backend.kill('SIGKILL');
      frontend.kill('SIGKILL');
    }, 2000);
  };

  process.on('SIGINT', shutdown);
  process.on('SIGTERM', shutdown);
}

main().catch(console.error);
```

#### [NEW] server/graceful-shutdown.ts - 优雅关闭模块

```typescript
import { Server as HttpServer } from 'http';
import { Server as SocketIOServer } from 'socket.io';
import { ChildProcess } from 'child_process';

/**
 * 优雅关闭管理器
 */
export class GracefulShutdown {
  private httpServer: HttpServer;
  private io: SocketIOServer | null = null;
  private activeProcesses = new Set<ChildProcess>();
  private shutdownTimeout = 5000; // 5秒强制关闭
  private isShuttingDown = false;

  constructor(httpServer: HttpServer) {
    this.httpServer = httpServer;
    this.setupHandlers();
  }

  setSocketIO(io: SocketIOServer) {
    this.io = io;
  }

  registerProcess(process: ChildProcess) {
    this.activeProcesses.add(process);
    process.on('exit', () => this.activeProcesses.delete(process));
  }

  private setupHandlers() {
    const signals = ['SIGTERM', 'SIGINT', 'SIGUSR2'];

    signals.forEach(signal => {
      process.on(signal as NodeJS.Signals, () => {
        console.log(`\n📴 收到 ${signal} 信号，开始优雅关闭...`);
        this.shutdown(signal);
      });
    });

    // 处理未捕获异常
    process.on('uncaughtException', (error) => {
      console.error('❌ 未捕获的异常:', error);
      this.shutdown('uncaughtException');
    });

    // 处理未处理的 Promise rejection
    process.on('unhandledRejection', (reason, promise) => {
      console.error('❌ 未处理的 Promise rejection:', reason);
      // 不直接关闭，记录日志
    });
  }

  private async shutdown(signal: string) {
    if (this.isShuttingDown) return;
    this.isShuttingDown = true;

    const deadline = Date.now() + this.shutdownTimeout;

    // 1. 关闭 Socket.IO 新连接
    if (this.io) {
      this.io.close();
      console.log('  ✓ Socket.IO 已关闭新连接');
    }

    // 2. 停止接受新的 HTTP 连接
    this.httpServer.close(() => {
      console.log('  ✓ HTTP 服务器已关闭');
    });

    // 3. 优雅关闭子进程
    console.log('  正在关闭子进程...');
    const processPromises = Array.from(this.activeProcesses).map(proc => {
      return new Promise<void>((resolve) => {
        const timer = setTimeout(() => {
          proc.kill('SIGKILL');
          resolve();
        }, 2000);

        proc.once('exit', () => {
          clearTimeout(timer);
          resolve();
        });

        proc.kill('SIGTERM');
      });
    });

    await Promise.all(processPromises);
    console.log(`  ✓ ${this.activeProcesses.size} 个子进程已关闭`);

    // 4. 等待待处理的请求（使用超时）
    const remaining = deadline - Date.now();
    if (remaining > 0) {
      await new Promise(r => setTimeout(r, Math.min(remaining, 1000)));
    }

    console.log(`✨ 优雅关闭完成 (${signal})`);
    process.exit(signal === 'uncaughtException' ? 1 : 0);
  }
}
```

```typescript
import express from 'express';
import fs from 'fs';
import path from 'path';

export function setupHealthCheck(app: express.Express) {
  const router = express.Router();

  router.get('/', (req, res) => {
    res.json({
      status: 'ok',
      timestamp: new Date().toISOString(),
      uptime: process.uptime(),
      memory: process.memoryUsage(),
      env: process.env.NODE_ENV || 'development'
    });
  });

  router.get('/ready', (req, res) => {
    // 检查关键依赖
    const checks = {
      config: checkFileExists('.env'),
      skills: checkDirectoryExists('skills')
    };

    const ready = Object.values(checks).every(v => v);

    res.status(ready ? 200 : 503).json({
      status: ready ? 'ready' : 'not_ready',
      checks
    });
  });

  router.get('/live', (req, res) => {
    res.json({ status: 'alive' });
  });

  app.use('/health', router);
}

function checkDirectoryExists(dir: string): boolean {
  try {
    return fs.existsSync(dir) && fs.statSync(dir).isDirectory();
  } catch {
    return false;
  }
}

function checkFileExists(file: string): boolean {
  try {
    return fs.existsSync(file) && fs.statSync(file).isFile();
  } catch {
    return false;
  }
}
```

## Feature 1: 评论/修改框 → 右侧固定面板 (Lovart 布局)

### 现状分析
- 当前 `ChatPanel` 是交付终端的浮动侧边栏，通过 `ChatToggleButton` 切换显示/隐藏
- `Phase1View` 的反馈输入框内嵌在卡片底部的 `textarea`
- `ChapterCard` (Phase2) 的评论框也是内嵌在每张卡片内
- 布局结构：`App.tsx` L145 `flex-1 flex overflow-hidden relative` → 左侧主内容 + 右侧可选 ChatPanel

### 设计方案
采用 **点击展开/收起** 模式（非始终显示），保持当前布局大体不变：

```
┌──────────────────────────────────────────────────────┐
│  Header (项目选择 / 模块切换)                          │
├────────────┬─────────────────────────┬──┐    ┌───────┤
│  ExpertNav │     主内容区 (弹性)       │按│ or │右侧面板│
│  (左侧导航) │  Phase1/Phase2/Phase3...│钮│    │360px  │
│            │                         │  │    │展开态  │
├────────────┴─────────────────────────┴──┘    └───────┤
│  StatusFooter                                         │
└──────────────────────────────────────────────────────┘
```

- **收起态**：右侧只显示一个竖条切换按钮（复用改造 `ChatToggleButton`），主内容区全宽
- **展开态**：右侧弹出 `RightPanel`（360px），主内容区自动压缩
- 切换动画：`transition-all duration-300`，流畅滑入滑出

## Feature 3: 多项目并行交付的业务逻辑

### 现状分析
- 当前服务器是单活项目模型：`currentProjectName` 全局变量
- `select-project` 事件切换时，服务端覆盖 `currentProjectName` + 重建所有 watcher
- 前端 `useDeliveryStore` 只维护一份 `state`
- 已有 SD-208 架构设计方案 (见 `docs/plans/2026-02-26_SD208_GoldenCrucible_Plan.md`)

### 设计方案（已确认：方案 B — 多浏览器窗口）

老卢通过手动管理浏览器 Tab 来区分项目，右上角项目名称 + 文档名称即可定位。
前端保持单项目模型不变，后端做最小改动支持多窗口并行。

**核心改动**：
- 后端 Socket.IO 引入 **Room 隔离**：每个 socket 连接 join 到其选择的 projectId Room
- `io.emit(...)` → `io.to(projectId).emit(...)` 确保事件不串台
- Watcher 改为按项目独立管理（支持多项目同时有 watcher）
- 前端无需改动

### 具体改动

#### [MODIFY] server/index.ts
- `currentProjectName` → `activeProjects: Map<string, { watcherCount: number }>`（引用计数）
- `select-project` 事件：`socket.join(projectId)`, 离开旧 Room
- 全局搜索 `io.emit('delivery-data', ...)` → `io.to(projectId).emit(...)`
- Watcher 从 "单项目覆盖" 改为 "多项目并存 + 引用计数"，当最后一个 socket 离开 Room 后才清理 watcher

## Feature 4: 全模块状态隔离持久化 (Bug 修复 → 架构升级)

### 设计原则

> [!IMPORTANT]
> **核心需求**：同一个项目内的每个专家模块（Director / Shorts / Marketing / Music / Thumbnail）各自独立持久化状态，切换模块时状态不丢失。用户可以在 Director Phase 2 审 B-Roll 时，随时切到 Shorts 工作，再切回来仍在 Phase 2。

### 现状分析（根因）

| 组件               | 受影响的 `useState`                    | 影响                                |
| ------------------ | -------------------------------------- | ----------------------------------- |
| `DirectorSection`  | `phase`, `concept`, `chapters`, `jobs` | ✅ 已修复（v3.8 Feature 4 原始方案） |
| `ShortsSection`    | `phase`, `scripts`, `renderUnits`      | ❌ 切模块后丢失                      |
| `MarketingSection` | `phase`                                | ❌ 切模块后回到 Phase 1              |
| `App.tsx`          | `activeExpertId`, `activeModule`       | ❌ 不持久化，刷新回默认              |

### 设计方案

**统一策略：所有 Section 组件改为受控组件，用 `delivery_store.json` 作为单一真相源**

```
delivery_store.json
├── projectId
├── activeModule        ← 新增
├── activeExpertId      ← 新增
├── modules
│   ├── director        ← 已有 + renderJobs
│   │   ├── phase
│   │   ├── conceptProposal
│   │   ├── isConceptApproved
│   │   ├── items (chapters)
│   │   └── renderJobs
│   ├── shorts (ShortsModule_V2)
│   │   ├── phase       ← 已有，但 ShortsSection 没从这读
│   │   ├── scripts
│   │   └── renderUnits
│   ├── marketing (MarketModule_V2)
│   │   ├── phase       ← 同上
│   │   └── titleTagSets
│   ├── music
│   │   ├── phase
│   │   └── ...
│   └── thumbnail
│       └── ...
```

### 具体改动

#### [MODIFY] src/types.ts
- `DirectorModule`: `phase` 改为 `number`，新增 `renderJobs?: any[]` ✅ 已完成
- `DeliveryState` 新增顶层字段：
  ```ts
  activeModule?: 'crucible' | 'delivery' | 'distribution';
  activeExpertId?: string;
  ```

#### [MODIFY] src/components/DirectorSection.tsx
- ✅ 已完成：从 `useState` 迁移到 `data` prop 读取

#### [MODIFY] src/components/ShortsSection.tsx
- `phase` → 从 `data.phase` 读取，`setPhase` → `onUpdate({...data, phase: x})`
- `scripts` → 从 `data.scripts` 读取
- `renderUnits` → 从 `data.renderUnits` 读取
- 仅保留 `lastModifiedId` 等纯 UI 瞬态属性为 `useState`

#### [MODIFY] src/components/MarketingSection.tsx
- `phase` → 从 `data.phase` 读取
- `isGenerating/isScoring/isConfirming` 保留为 `useState`（纯 UI 加载态）

#### [MODIFY] src/App.tsx
- `activeModule` 从 `useState` 改为从 `state.activeModule` 读取
- `activeExpertId` 从 `useState` 改为从 `state.activeExpertId` 读取
- 切换操作调用 `updateState({...state, activeModule: x})` 写回 store
- 这样刷新页面或切换项目后，自动恢复到上次的模块+专家

#### [MODIFY] server/index.ts
- `ensureDeliveryFile` 初始结构增加 `activeModule: 'delivery'`, `activeExpertId: 'Director'`
- `director` 模块增加 `renderJobs: []` ✅ 已完成
